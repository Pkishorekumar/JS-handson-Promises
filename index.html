Q.4:-"Create a promise function accepting an argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed 
then it should go to reject the state and catch the error and print Promise Rejected "

function myPromise(a) {
    return new Promise((resolve,reject)=>{
        if(a==="yes"){
            resolve("Promise Resolved");
        }
        else{
            reject("Promise Rejected");
        }
    });
}
myPromise("yes")
.then(result => console.log(result))
.catch(error => console.log(error));

        Q.9:-Create examples to explain promise.all function


        let a = Promise.resolve("PrepBytes");
        let b = Promise.resolve("Elevation");
        let c = Promise.resolve("Academy");

        Promise.all([a, b, c]).then(values => {
        console.log(values);// Output: [ 'PrepBytes', 'Elevation', 'Academy' ]
        });

        let d = Promise.reject("PrepBytes");
        let e = Promise.reject("Elevation");
        let f = Promise.reject("Academy");

        Promise.all([d, e, f]).catch(values => {
        console.log(values);// Output: PrepBytes
        });
